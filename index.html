let bubbles = [];

  createCanvas(600, 400);
}

function mouseDragged() {
  let r = random(10, 50);

bubbles[i].move();
bubbles[i].show();
  }
}

class Bubble {
  constructor(x, y, r) {
this.x = x;
this.y = y;
this.r = r;
  }

  move() {
this.x = this.x + random(-5, 5);
this.y = this.y + random(-5, 5);
  }

  show() {
stroke(255);
strokeWeight(4);
noFill();
ellipse(this.x, this.y, this.r * 2);

car = {

var car1;
var car2; // Two objects!

function setup() {
  createCanvas(480, 400);
  car1 = new Car(color(127),0,100,2); // Parameters go inside the parentheses when the object is constructed.
  car2 = new Car(color(175),0,300,1);
}

function draw() {
  background(51);
  car1.move();
  car1.display();
  car2.move();
  car2.display2();
}

// Even though there are multiple objects, we still only need one constructor function. 
// No matter how many cookies we make, only one cookie cutter is needed.Isn’t object-oriented programming swell?
function Car(tempC, tempXpos, tempYpos, tempXspeed) { // The Constructor is defined with arguments.
  this.c = tempC;
  this.xpos = tempXpos;
  this.ypos = tempYpos;
  this.xspeed = tempXspeed;

  this.display = function() {
stroke(0);
fill(this.c);
rectMode(CENTER);
rect(this.xpos,this.ypos,20,10);
  }

  this.display2 = function() {
stroke(0);
fill(this.c);
rectMode(CENTER);
rect(this.xpos,this.ypos,20,10);
  }

  this.move = function() {
this.xpos = this.xpos + this.xspeed;
if (this.xpos > width) {
  this.xpos = 0;
}
  }
}

Conditional Statements

When the test, what is inside the parentheses, passes–it becomes true–then the statement between the curly braces will run:

if (test)
statements
}

For example, in the following code, the point will continue to move along the x axis as long as test inside the if( test ) does NOT pass. 

function draw() {
strokeWeight(3);
point(x+=1, 25);
if (test) {
  statements
}
}

Once the test inside if( test ) passes, the statement that is inside the { } curly braces runs.

Take a look at this diagram:

Again, until the test does NOT pass, according to the diagram, function draw() continues to run.

Let’s setup a test. 

function draw() {
strokeWeight(3);
point(x+=1, 25);
if ( x > 240 ) {
  x = 0;    }
}

What will happen here? The point will move along the x axis until it reaches x = 240, which makes that test false ( does not pass: 240 > 240 is NOT true. ), therefore the statement inside the { } curly braces executes ( runs ) and x is repositioned at x = 0.

Try it out.

See me here

So, again, when the if()test is false, the function draw(); continues to runs. When the if()test is true, the code inside the { }curly braces runs. The computer determines whether the test is true or false by evaluating the expression inside the parentheses.

We have at present an if() condition that loops the point for a reason. It is important for you to understand what makes this effect. Note: Each time x is reset back to 0, the test inside if() continues to be false ( not pass ), therefore the function draw runs over and over again, executing if() over and over again. The test in if() becomes true ( passes ) only for an instance, when it reaches 241, and when it does, it allows the statement inside the { } curly braces to run, resenting x to x = 0, which in turn makes the point move again until it reaches 241 again, making the if() test pass again, and again.

If() rule:

if (condition)
block of code to be executed if the condition is true
}

The else statement:

if (condition)
block of code to be executed if the condition is true
} else
block of code to be executed if the condition is false
}

How can we put to use these two system variables? As the documentation explains, these two system variables store the width and height of canvas created by the two parameters in createCanvas( ). The values you placed in each parameter are stored in width and height.

What is this doing? How would you read it and explain it?

Keep in mind that a single = in programming represents assignment, not equality. For equality the double == is used.

For example: x = 10 means “Assign ( store or save ) to variable x the integer 10.” Whereas x == 10 means “Variable x equals integer 10.”
