Object Oriented Programming

Creating Classes

OOP, Object Oriented Programming, is what’s at the heart of all advance code. If you desire to develop into skilled programmer you will need to master OOP.

This is an introductory programming course so we will not have much time to dedicate to OOP, but we should make the attempt at getting our feet wet, so to speak.

Take a look at this video. It covers what are called JavaScript data objects:

We will be studying the following two programs. Take a look at this code:

var car; // Declare car object as a global variable.
function setup() {
  createCanvas(480, 270);
  // Initialize Car object
car = {
c: color(175),
xpos: width/2,
ypos: height/2,
xspeed: 3,
display: function() { // Function.
  // The car is just a square
  rectMode(CENTER);
  stroke(0);
  fill(this.c);
  rect(this.xpos,this.ypos,20,10);
},
move: function() { // Function.
  this.xpos = this.xpos + this.xspeed;
  if (this.xpos > width) {
this.xpos = 0;
  }
}    
  };
}
function draw() {
  background(100);
  // Operate Car object.
  car.move(); // Operate the car object in draw( ) by calling object methods using the dots syntax.
  car.display();
}

Now take a look at this code:

var car1;
var car2; // Two objects!

function setup() {
  createCanvas(480, 400);
  car1 = new Car(color(127),0,100,2); // Parameters go inside the parentheses when the object is constructed.
  car2 = new Car(color(175),0,300,1);
}

function draw() {
  background(51);
  car1.move();
  car1.display();
  car2.move();
  car2.display2();
}

// Even though there are multiple objects, we still only need one constructor function. 
// No matter how many cookies we make, only one cookie cutter is needed.Isn’t object-oriented programming swell?
function Car(tempC, tempXpos, tempYpos, tempXspeed) { // The Constructor is defined with arguments.
  this.c = tempC;
  this.xpos = tempXpos;
  this.ypos = tempYpos;
  this.xspeed = tempXspeed;

  this.display = function() {
stroke(0);
fill(this.c);
rectMode(CENTER);
rect(this.xpos,this.ypos,20,10);
  }

  this.display2 = function() {
stroke(0);
fill(this.c);
rectMode(CENTER);
rect(this.xpos,this.ypos,20,10);
  }

  this.move = function() {
this.xpos = this.xpos + this.xspeed;
if (this.xpos > width) {
  this.xpos = 0;
}
  }
}
